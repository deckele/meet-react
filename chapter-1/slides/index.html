HTML Components

Native API

document.createElement(tagName[, options])

// example
document.body.onload = addElement;
function addElement () {
  const title = document.createElement("h1"); 
  const content = document.createTextNode("Hello there!"); 
  title.appendChild(content);  

  const appRoot = document.getElementById("root"); 
  appRoot.appendChild(title); 
}

Meet React.js

Creating React component


React.createElement(
  type,
  [props],
  [...children]
)

// example
React.createElement('h1', {}, 'Hello React!');

Render using ReactDOM


ReactDOM.render(element, container[, callback])

// example
const Title = React.createElement('h1', {}, 'Hello React!');
ReactDOM.render(Title, document.getElementById('root'));

Communicate Between Components

Component props


const Header = (props) => (<h1>{props.title}</h1>);

// Usage:
<Header title="Hello React!" />

// Result:
<h1>Hello React!</h1>


props are read-only!!!

Component props


const Header = ({title}) => (<h1>{title}</h1>);

// Usage:
<Header title="Hello React!" />

// Result:
<h1>Hello React!</h1>


props are read-only!!!

Component events


const Header = ({title, onPress}) => 
                    (<h1 onClick={onPress}>{title}</h1>);

// Usage:
<Header onPress={() => alert('clicked!')} title="Press here" />


props are read-only!!!

Components composition

const ComponentA = (props) => (<div>Content is: {props.children}</div>);
const Title = () => (<h1>Hello React!</h1>);

// Usage:
<ComponentA>
  <Title />
</ComponentA>

// Result:
<div>
    Content is: <h1>Hello React!</h1>
</div>

Meet React.ts

Function component

import * as React from 'react';

interface HeaderProps {
  title: string; 
  onClick: () => void;
}

const Header: React.SFC<HeaderProps> = ({title, onClick}) => 
                          (<h1 onClick={onClick}>{title}</h1>);

Class component

import * as React from 'react';

interface HeaderProps {
  title: string; 
  onClick: () => void;
}

class Header extends React.Component<HeaderProps>{
  render() {
      return 
      (<h1 onClick={this.props.onClick}>{this.props.title}</h1>); 
  }
}
