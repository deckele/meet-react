export const selectCard = (id: number) => ({
    type: SELECT_CARD,
    payload: { id }
});

Reducers

The only way to modify the state
const gameReducer: Reducer = (state = initialState, action) => {
    switch (action.type) {
        case SELECT_CARD:
            return {
                ...state,
                selectedCard: action.payload,
            };
        default:
            return state;
    }
};
State is immutable!

const startGameMW: Middleware = (store) => (next) => (action) => {
  if (action.type === START_GAME) {
    store.dispatch(actions.shuffleCards());
  } else {
    next(action);
  }
}
